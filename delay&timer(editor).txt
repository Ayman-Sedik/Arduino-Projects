#define led 13
#define pb 2
#define gnd 4
int i=0;
long t=0,duration=0;
void setup()‬
{
 Serial.begin(9600); ‬
 pinMode(led,OUTPUT);
 pinMode(pb,INPUT_PULLUP);
 pinMode(gnd,OUTPUT);
 digitalWrite(gnd,LOW);
 t=millis();
  
}
void loop()
{
  duration=millis()-t;//انا كدا بحسب التايم من اول التي لحد الديوريشن
  if(duration>=1000)
  {
    Serial.println(i);//ده معناه لما يعدي الثانيه من اول الفويد سيتاب لحد الفويد لوب
    i++;
    t=millis();//لازم اصفر قيمه التي علشان يطبع كل ثانيه يعني هيحط في التي قيمه الميلز من اول مادخل ال اف بضي يعني هيحط في التي التايم اللي في الميلز فلما يبدا مقارنه من المره الجايه يبدا ع اساس انه واقف تحت مش فوق
    }
   if(digitalRead(pb)==LOW)
   {
    digitalWrite(led,HIGH);
    }else digitalWrite(led,LOW); 
    
   
}
