//pull down resistor
#define sw 2
#define gnd 4
void setup() {
 Serial.begin(9600);
 pinMode(sw,INPUT); 
 pinMode(gnd,OUTPUT);
 digitalWrite(gnd,LOW); 
}

void loop() {
Serial.println(digitalRead(sw));
}
//pull up resistor  
#define sw 2
#define led LED_BUILTIN
void setup() {
 Serial.begin(9600);
 pinMode(sw,INPUT); 
 pinMode(led,OUTPUT);
 }

void loop() {
  if(digitalRead(sw)==LOW) 
  {
    delay(10);
    digitalWrite(led,HIGH);
    }else digitalWrite(led,LOW);
  }
 
#define sw 2
#define gnd 4
#define led LED_BUILTIN
void setup() {
 Serial.begin(9600);
 pinMode(sw,INPUT_PULLUP);  
 pinMode(led,OUTPUT);
 pinMode(gnd,OUTPUT);
 digitalWrite(gnd,LOW);
 }

void loop() {
  if(digitalRead(sw)==LOW) 
  {
    delay(10);
    digitalWrite(led,HIGH);
    }else digitalWrite(led,LOW);
  }
