#define led 13
#define pb 2
#define gnd 4
int i=0;
long t=0,duration=0;
void setup()‬
{
 Serial.begin(9600); ‬
 pinMode(led,OUTPUT);
 pinMode(pb,INPUT_PULLUP);
 pinMode(gnd,OUTPUT);
 digitalWrite(gnd,LOW);
 t=millis();
  
}
void loop()
{
  duration=millis()-t; 
  if(duration>=1000)
  {
    Serial.println(i); 
    i++;
    t=millis(); 
    }
   if(digitalRead(pb)==LOW)
   {
    digitalWrite(led,HIGH);
    }else digitalWrite(led,LOW); 
    
}
